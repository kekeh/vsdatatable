// data operation (add, edit or delete) callback

    if (phase === vsdatatableConfig.OPER_PHASE_END
        && operation === vsdatatableConfig.OPER_ADD) {
        // add new item to the end of the array
    }
    else if (phase === vsdatatableConfig.OPER_PHASE_END
        && operation === vsdatatableConfig.OPER_EDIT) {
        // replace the old item with the new one
    }
    else if (phase === vsdatatableConfig.OPER_PHASE_END
        && operation === vsdatatableConfig.OPER_DELETE) {
        // delete the item from the array
    }
};
